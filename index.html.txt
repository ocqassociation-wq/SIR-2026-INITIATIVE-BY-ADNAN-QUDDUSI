<script>
    const scriptUrl = 'https://script.google.com/macros/s/AKfycbwXU1HhDcmxUxMjnzWX63agUEqAQrC-CGwEqXAfKU411cgFtd5e5GcAVqswsmZYy1L0dw/exec';
    let bloDatabase = {};

    // 1. Fetch BLO data from your Google Sheet automatically
    async function loadBLOs() {
        try {
            const response = await fetch(scriptUrl);
            bloDatabase = await response.json();
            document.getElementById('bloDataText').innerText = "Database Ready. Enter Part Number above.";
        } catch (e) {
            document.getElementById('bloDataText').innerText = "Error loading database. Please refresh.";
        }
    }
    loadBLOs();

    // 2. Search Logic
    document.getElementById('partSearch').addEventListener('input', function() {
        const val = this.value;
        const resDiv = document.getElementById('bloResult');
        const textDiv = document.getElementById('bloDataText');
        if (bloDatabase[val]) {
            resDiv.style.display = 'block';
            const b = bloDatabase[val];
            textDiv.innerHTML = `👤 <b>BLO:</b> ${b.name}<br>📞 <b>Mobile:</b> <a href="tel:${b.mobile}">${b.mobile}</a><br>📍 <b>Loc:</b> ${b.loc}`;
            document.getElementById('formPartNo').value = val;
            document.getElementById('formBloName').value = b.name;
        } else { resDiv.style.display = 'none'; }
    });

    // 3. Form Submit
    document.getElementById('sirForm').addEventListener('submit', function(e) {
        e.preventDefault();
        const btn = document.getElementById('submitBtn');
        btn.innerText = "Sending..."; btn.disabled = true;
        fetch(scriptUrl, { method: 'POST', mode: 'no-cors', body: JSON.stringify(Object.fromEntries(new FormData(this).entries())) })
        .then(() => { alert("Success! Check your WhatsApp."); this.reset(); btn.innerText = "Submit Details"; btn.disabled = false; });
    });
</script>
